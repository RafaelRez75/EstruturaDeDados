   i = j - 1

     v[i + 1] = v[i]
     i = i - 1
   v[i + 1] = x
 return v

    m = min(v)
    resp.append(m)
    v.remove(m)

def merge(e, d):
    r = []
    i, j = 0, 0
    while i < len(e) and j < len(d):
        if e[i] <= d[j]:
            r.append(e[i])
            i += 1
        else:
            r.append(d[j])
            j += 1
    r += e[i:]
    r += d[j:]
    return r

def mergesort(v):
    if len(v) <= 1:
        return v
    else:
        m = len(v) // 2
        e = mergesort(v[:m])
        d = mergesort(v[m:])
        return merge(e, d)

print (mergesort([3, 4, 8, 0, 6, 7, 4, 2, 1, 9, 4, 5]))
